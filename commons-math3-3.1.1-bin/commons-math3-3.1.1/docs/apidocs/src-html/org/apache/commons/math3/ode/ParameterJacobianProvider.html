017    package org.apache.commons.math3.ode;
018    
019    import org.apache.commons.math3.exception.DimensionMismatchException;
020    import org.apache.commons.math3.exception.MaxCountExceededException;
021    
022    /** Interface to compute exactly Jacobian matrix for some parameter
023     *  when computing {@link JacobianMatrices partial derivatives equations}.
024     *
025     * @version $Id: ParameterJacobianProvider.java 1416643 2012-12-03 19:37:14Z tn $
026     * @since 3.0
027     */
028    public interface ParameterJacobianProvider extends Parameterizable {
029    
030        /** Compute the Jacobian matrix of ODE with respect to one parameter.
031         * <p>If the parameter does not belong to the collection returned by
032         * {@link #getParametersNames()}, the Jacobian will be set to 0,
033         * but no errors will be triggered.</p>
034         * @param t current value of the independent <I>time</I> variable
035         * @param y array containing the current value of the main state vector
036         * @param yDot array containing the current value of the time derivative
037         * of the main state vector
038         * @param paramName name of the parameter to consider
039         * @param dFdP placeholder array where to put the Jacobian matrix of the
040         * ODE with respect to the parameter
041         * @exception MaxCountExceededException if the number of functions evaluations is exceeded
042         * @exception DimensionMismatchException if arrays dimensions do not match equations settings
043         * @exception UnknownParameterException if the parameter is not supported
044         */
045        void computeParameterJacobian(double t, double[] y, double[] yDot,
046                                      String paramName, double[] dFdP)
047            throws DimensionMismatchException, MaxCountExceededException, UnknownParameterException;
048    
049    }
