018    package org.apache.commons.math3.optimization;
019    
020    import org.apache.commons.math3.analysis.MultivariateFunction;
021    
022    /**
023     * This interface is mainly intended to enforce the internal coherence of
024     * Commons-FastMath. Users of the API are advised to base their code on
025     * the following interfaces:
026     * <ul>
027     *  <li>{@link org.apache.commons.math3.optimization.MultivariateOptimizer}</li>
028     *  <li>{@link org.apache.commons.math3.optimization.MultivariateDifferentiableOptimizer}</li>
029     * </ul>
030     *
031     * @param <FUNC> Type of the objective function to be optimized.
032     *
033     * @version $Id: BaseMultivariateOptimizer.java 1422230 2012-12-15 12:11:13Z erans $
034     * @deprecated As of 3.1 (to be removed in 4.0).
035     * @since 3.0
036     */
037    @Deprecated
038    public interface BaseMultivariateOptimizer<FUNC extends MultivariateFunction>
039        extends BaseOptimizer<PointValuePair> {
040        /**
041         * Optimize an objective function.
042         *
043         * @param f Objective function.
044         * @param goalType Type of optimization goal: either
045         * {@link GoalType#MAXIMIZE} or {@link GoalType#MINIMIZE}.
046         * @param startPoint Start point for optimization.
047         * @param maxEval Maximum number of function evaluations.
048         * @return the point/value pair giving the optimal value for objective
049         * function.
050         * @throws org.apache.commons.math3.exception.DimensionMismatchException
051         * if the start point dimension is wrong.
052         * @throws org.apache.commons.math3.exception.TooManyEvaluationsException
053         * if the maximal number of evaluations is exceeded.
054         * @throws org.apache.commons.math3.exception.NullArgumentException if
055         * any argument is {@code null}.
056         * @deprecated As of 3.1. In 4.0, it will be replaced by the declaration
057         * corresponding to this {@link org.apache.commons.math3.optimization.direct.BaseAbstractMultivariateOptimizer#optimize(int,MultivariateFunction,GoalType,OptimizationData[]) method}.
058         */
059        @Deprecated
060        PointValuePair optimize(int maxEval, FUNC f, GoalType goalType,
061                                double[] startPoint);
062    }
