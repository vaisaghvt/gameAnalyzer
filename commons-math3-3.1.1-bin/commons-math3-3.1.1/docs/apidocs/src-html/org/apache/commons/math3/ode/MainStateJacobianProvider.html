017    package org.apache.commons.math3.ode;
018    
019    import org.apache.commons.math3.exception.DimensionMismatchException;
020    import org.apache.commons.math3.exception.MaxCountExceededException;
021    
022    /** Interface expanding {@link FirstOrderDifferentialEquations first order
023     *  differential equations} in order to compute exactly the main state jacobian
024     *  matrix for {@link JacobianMatrices partial derivatives equations}.
025     *
026     * @version $Id: MainStateJacobianProvider.java 1416643 2012-12-03 19:37:14Z tn $
027     * @since 3.0
028     */
029    public interface MainStateJacobianProvider extends FirstOrderDifferentialEquations {
030    
031        /** Compute the jacobian matrix of ODE with respect to main state.
032         * @param t current value of the independent <I>time</I> variable
033         * @param y array containing the current value of the main state vector
034         * @param yDot array containing the current value of the time derivative of the main state vector
035         * @param dFdY placeholder array where to put the jacobian matrix of the ODE w.r.t. the main state vector
036         * @exception MaxCountExceededException if the number of functions evaluations is exceeded
037         * @exception DimensionMismatchException if arrays dimensions do not match equations settings
038         */
039        void computeMainStateJacobian(double t, double[] y, double[] yDot, double[][] dFdY)
040            throws MaxCountExceededException, DimensionMismatchException;
041    
042    }
