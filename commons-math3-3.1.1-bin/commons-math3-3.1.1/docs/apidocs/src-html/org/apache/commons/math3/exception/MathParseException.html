017    package org.apache.commons.math3.exception;
018    
019    import org.apache.commons.math3.exception.util.ExceptionContextProvider;
020    import org.apache.commons.math3.exception.util.LocalizedFormats;
021    
022    /**
023     * Class to signal parse failures.
024     *
025     * @since 2.2
026     * @version $Id: MathParseException.java 1364378 2012-07-22 17:42:38Z tn $
027     */
028    public class MathParseException extends MathIllegalStateException
029        implements ExceptionContextProvider {
030        /** Serializable version Id. */
031        private static final long serialVersionUID = -6024911025449780478L;
032    
033        /**
034         * @param wrong Bad string representation of the object.
035         * @param position Index, in the {@code wrong} string, that caused the
036         * parsing to fail.
037         * @param type Class of the object supposedly represented by the
038         * {@code wrong} string.
039         */
040        public MathParseException(String wrong,
041                                  int position,
042                                  Class<?> type) {
043            getContext().addMessage(LocalizedFormats.CANNOT_PARSE_AS_TYPE,
044                                    wrong, Integer.valueOf(position), type.getName());
045        }
046    
047        /**
048         * @param wrong Bad string representation of the object.
049         * @param position Index, in the {@code wrong} string, that caused the
050         * parsing to fail.
051         */
052        public MathParseException(String wrong,
053                                  int position) {
054            getContext().addMessage(LocalizedFormats.CANNOT_PARSE,
055                                    wrong, Integer.valueOf(position));
056        }
057    }
